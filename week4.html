첫 마운트 — 컴포넌트를 화면에 한 번 붙이기

무슨 단계인가: “화면 조각”을 나타내는 함수(renderFn)를 실행해 가상 트리(VNode)를 만든 뒤, Week 3의 vnodeToDom으로 진짜 DOM을 만들어 컨테이너에 넣습니다. 이걸 묶는 진입점이 FunctionComponent의 mount(container)입니다. 아래 데모 박스 안에 흐름이 그림으로 있습니다.

훅 없이 h / hText만 사용합니다. 버튼으로 바뀌는 건 2단계부터입니다.

실제 코드 보기 · 1단계(첫 마운트)

container는 서버에서 오는 값이 아니라, 이 페이지의 DOM에서 document.getElementById('container-week4-1-mount')로 찾은 div 요소입니다. 그 요소를 mount(container)에 넘겨서 첫 렌더 결과를 붙입니다.

js/week4-demo.js (핵심) 
mountIf(
  document.getElementById('container-week4-1-mount'), // HTML에 있는 마운트 대상 div
  function Week4Phase1() { // 렌더 함수(renderFn)
    return h('div', { class: 'week4-app week4-app--phase1' }, [ /* ... */ ]); // VNode 트리 반환
  }
);
js/mini-react.js (핵심) 
FunctionComponent.prototype.mount = function (container) {
  this.container = container; // 이후 update에서도 같은 컨테이너 사용
  activeComponent = this; // 지금 훅이 바라볼 컴포넌트 지정
  var vnode = this.renderFn(); // renderFn 실행 -> 새 VNode
  activeComponent = null; // 렌더 끝났으니 활성 표시 해제
  container.replaceChildren(); // 컨테이너 비우기(첫 마운트)
  var n = vnodeToDom(vnode); // VNode -> 실제 DOM 노드
  if (n) container.appendChild(n); // 화면에 붙이기
};
2

상태 — 바뀌면 같은 규칙으로 다시 그리기

무슨 단계인가: useState로 “기억할 값”과 “바꾸는 함수”를 둡니다. setter(setState)가 호출되면 나중에 update()가 돌아가고, 그 안에서 diffVNode → applyPatches로 DOM만 최소한 고칩니다. (여러 번 setter가 나와도 microtask 한 번에 묶일 수 있음.)

실제 코드 보기 · 2단계(useState)

버튼 클릭 → setN 호출 → scheduleUpdate가 microtask에 update() 예약 → diffVNode / applyPatches로 최소 변경합니다.

js/week4-demo.js (핵심) 
var st = useState(0); // [값, setter]
var n = st[0]; // 현재 상태값
var setN = st[1]; // 상태 갱신 함수
onClick: function () {
  setN(function (x) { return x + 1; }); // 이전값 기반 +1
};
js/mini-react.js (핵심) 
function useState(initial) {
  var i = comp.hookIndex++; // 이번 렌더의 훅 슬롯 번호
  if (!comp.hooks[i]) comp.hooks[i] = { type: 'state', value: initial }; // 첫 렌더 초기화
  var hook = comp.hooks[i]; // 같은 슬롯 재사용
  function setState(next) { hook.value = next; scheduleUpdate(comp); } // 값 갱신 + update 예약
  return [hook.value, setState]; // [현재값, setter]
}
3

기억해 둔 계산 — 의존할 때만 다시 계산

무슨 단계인가: useMemo는 비싼 계산이나 파생 값을 의존 배열(deps)이 바뀔 때만 다시 실행합니다. count만 deps에 넣고 tick은 빼 두면, tick만 올려서 리렌더가 나도 캐시된 결과가 유지되는지 확인할 수 있습니다.

실제 코드 보기 · 3단계(useMemo)

derived는 [count]가 바뀔 때만 다시 계산합니다. tick만 바뀌면 리렌더는 나도 memo 값은 그대로입니다.

js/week4-demo.js (핵심) 
var derived = useMemo(
  function () { return count * 10 + 1; }, // 계산식(factory)
  [count] // count가 바뀔 때만 factory 재실행
);
js/mini-react.js (핵심) 
function useMemo(factory, deps) {
  var hook = comp.hooks[i]; // 이전 memo 슬롯
  if (!hook || !shallowDepsEqual(hook.deps, deps)) { // 첫 렌더 또는 deps 변경
    var computed = factory(); // 다시 계산
    comp.hooks[i] = { type: 'memo', deps: deps, value: computed }; // 캐시 저장
    return computed;
  }
  return hook.value; // deps 같으면 캐시 재사용
}
4

그린 뒤에 할 일 — 부수 효과

무슨 단계인가: useEffect는 “VNode→DOM 반영”이 끝난 다음(microtask)에 돌아갑니다. 화면을 다시 그리는 게 아니라, 로그·외부 DOM·구독 같은 부수 효과를 두는 자리입니다. deps가 바뀌면 cleanup 후 다시 실행됩니다. 아래 문구는 theme / msg에 맞춰 갱신됩니다.

아래 로그는 최신 항목이 위로 쌓입니다. 버튼을 눌러보면 render → cleanup(이전) → effect(현재) 순서를 직관적으로 볼 수 있습니다.

    실제 코드 보기 · 4단계(useEffect)

    effect 본문은 렌더 함수 안에서 등록만 하고, 실제 실행은 flushEffects()의 microtask에서 처리됩니다.

    js/week4-demo.js (핵심) 
    useEffect(function () {
  var el = document.getElementById('output-week4-4-effect'); // 외부 DOM 참조
  if (el) el.textContent = '[effect] theme=' + theme + ', msg=' + msg; // 부수 효과
  return function () {}; // cleanup 자리
}, [theme, msg]); // deps 바뀔 때만 다시 실행
    js/mini-react.js (핵심) 
    FunctionComponent.prototype.flushEffects = function () {
  queueMicrotask(function () { // 커밋 직후 실행
    if (hook.pendingRun) { // 이번에 돌아야 하는 effect만
      if (hook.cleanup) hook.cleanup(); // 이전 cleanup 먼저
      var ret = hook.fn(); // effect 본문 실행
      if (typeof ret === 'function') hook.cleanup = ret; // 다음 cleanup 저장
    }
  });
};

    자식은 넘겨받기만 — 상태는 위(루트)에만

    무슨 단계인가: Header 같은 조각은 h(Header, props)로만 그립니다. 자식 함수는 props만 받아 VNode를 반환하고, useState 등은 이 데모의 루트에만 둡니다. 이걸 관례적으로 lifting state up(상태 끌어올리기)라고 부릅니다.

    실제 코드 보기 · 5단계(무상태 자식 + 상태 끌어올리기)

    자식 함수 Header는 props만 받고 VNode를 반환합니다. 상태는 루트(Week4Phase5)에만 두고 자식에게 props로 내려줍니다.

    js/week4-demo.js (핵심) 
    function Header(props) {
  return h('header', { class: 'week4-panel__head' }, /* ... */); // props 기반 표시 전용
}

return h('div', { class: 'week4-app' }, [
  h(Header, { title: '5단계 · Stateless Header', subtitle: 'props만 받음 · 훅 없음' }) // 자식 호출
]);
    js/mini-react.js (핵심) 
    function h(type, props) {
  var childArgs = Array.prototype.slice.call(arguments, 2); // 3번째 인자부터 children
  props = props == null ? {} : assignProps({}, props); // props 정규화/복사

  if (typeof type === 'function') { // 함수형 자식 컴포넌트 분기
    var prev = activeComponent; // 부모 컨텍스트 저장
    activeComponent = null; // 자식에서 훅 사용 차단
    try { return type(props); } // 자식 함수 실행
    finally { activeComponent = prev; } // 부모 컨텍스트 복구
  }

  return {
    type: type, // 일반 태그 타입
    props: props, // 속성
    children: normalizeChildren(childArgs), // 자식 배열 정규화
  };
}
